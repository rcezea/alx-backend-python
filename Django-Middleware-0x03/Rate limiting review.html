✅ Refactored Middleware: OffensiveLanguageMiddleware

python
import logging
from datetime import timedelta

from django.utils import timezone
from django.http import JsonResponse
from django.conf import settings
from django.core.cache import cache
from rest_framework import status

logger = logging.getLogger(__name__)


class OffensiveLanguageMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response

        # Defaults; can override via Django settings
        self.rate_limit_window = getattr(settings, 'RATE_LIMIT_WINDOW', 10)  # in minutes
        self.rate_limit_max = getattr(settings, 'RATE_LIMIT_MAX', 5)
        self.rate_limit_path = getattr(settings, 'RATE_LIMIT_PATH', '/api/messages/')

    def __call__(self, request):
        if request.method == 'POST' and request.path == self.rate_limit_path:
            client_ip = get_client_ip(request)
            cache_key = f"msg_rate_limit_{client_ip}"
            record = cache.get(cache_key)

            current_time = timezone.now()

            if record:
                window_start = record['start_time']
                count = record['count']
                if current_time - window_start <= timedelta(minutes=self.rate_limit_window):
                    if count >= self.rate_limit_max:
                        logger.warning(f"Rate limit exceeded for IP: {client_ip}")
                        return JsonResponse(
                            data={"error": "Message Limit Reached."},
                            status=status.HTTP_403_FORBIDDEN
                        )
                    else:
                        record['count'] += 1
                else:
                    # Reset window
                    record = {'count': 1, 'start_time': current_time}
            else:
                # First time request
                record = {'count': 1, 'start_time': current_time}

            cache.set(cache_key, record, timeout=self.rate_limit_window * 60)

        return self.get_response(request)


def get_client_ip(request):
    """Return the real client IP address from request headers."""
    x_forwarded_for = request.META.get('HTTP_X_FORWARDED_FOR')
    if x_forwarded_for:
        ip = x_forwarded_for.split(',')[0].strip()
    else:
        ip = request.META.get('REMOTE_ADDR')
    return ip
